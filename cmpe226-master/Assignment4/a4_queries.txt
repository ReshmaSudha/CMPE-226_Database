FROM CUSTOMER A, `ORDER` B, `PAYMENT`C

SELECT C.BrandName, B.ProuctName, B.Model, B.Description, B.Price, D.ReviewId 
FROM SUBCATEGORY A, PRODUCT B, BRAND C, REVIEW D
